##节流函数 throttle
###方案一

防抖函数 debounce

方案一

function debounce(fn, time){
    let timer = null;
    return function(...args) {
        if(!timer) {
            timer = setTimeout(() => {
                fn.call(this, args)
            }, time)   
        }
        clearTimeout(timer)
    }

}
